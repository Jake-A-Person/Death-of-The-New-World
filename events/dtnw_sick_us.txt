dtnw_sick_us.21{

	type = country_event
	placement = root
	title = dtnw_sick_us.21.t
	desc = dtnw_sick_us.21.d
	flavor = dtnw_sick_us.21.f

	event_image = {
		video = "gfx/event_pictures/southamerica_aristocrats.bk2"
	}

	duration = 1

	trigger = {

	}

	immediate = {
	}

	#has slaves banned
	option = {	
		trigger = {
			has_law = law_type:law_slavery_banned
		}
		name = dtnw_sick_us.21.a
		default_option = yes
	}


	#civl war
	option = {	
		trigger = {
			not = {
				has_law = law_type:law_slavery_banned
			}
		}
		name = dtnw_sick_us.21.b
		default_option = yes
		#not an error just new in 1.7
		add_liberty_desire = 100
		change_relations = {
			country = overlord
			value = -100
		}
		if = {
			limit = {
					NOT = {has_variable = sick_us_civl_war}
				}
		add_journal_entry = {type = je_civl_war_start}
		}
	}

	#ban slaves
	option = {	
		trigger = {
			not = {
				has_law = law_type:law_slavery_banned
			}
		}
		name = dtnw_sick_us.21.c
	}
}
