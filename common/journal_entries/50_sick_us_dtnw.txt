je_civil_war_start = {
	icon = "gfx/interface/icons/event_icons/event_election.dds"
	
	group = je_group_internal_affairs
	

	is_shown_when_inactive = {
		not = {
			has_modifier = civil_war_nation
		}
		any_subject_or_below = {
			has_modifier = civil_war_nation
			count > 0
		}
	}

	possible = {
		not = {
			has_modifier = civil_war_nation
		}
		any_subject_or_below = {
			has_modifier = civil_war_nation
			count > 0
		}
	}

	immediate = {	
		
	}

	timeout = 42 # weeks

	on_timeout = {
		if = {
			limit = {
				any_subject_or_below = {
					has_modifier = civil_war_nation
					is_player = yes
				}
			}
			random_subject_or_below = {
				limit = {
					has_modifier = civil_war_nation
					is_player = yes
				}
				save_scope_as = civil_war_usa_free_leader
			}
		}
		else = {
			ordered_subject_or_below = {
				limit = {
					has_modifier = civil_war_nation
				}
				order_by = gdp
				save_scope_as = civil_war_usa_free_leader
			}
		}

		scope:civil_war_usa_free_leader = {
			make_independent = yes
		}

		every_subject_or_below = {
			limit = {
				has_modifier = civil_war_nation
				not = {scope:civil_war_usa_free_leader=this}
			}
			make_independent = yes
			scope:civil_war_usa_free_leader = {
				create_diplomatic_pact = {
					country = prev
					type = puppet
				}
			}
		}

		create_diplomatic_play = {
			type = dp_make_protectorate
			target_country = scope:civil_war_usa_free_leader
		}
		add_journal_entry = {
			type = je_civil_war_end
		}
	}
}

je_civil_war_end = {
	icon = "gfx/interface/icons/event_icons/event_election.dds"
	
	group = je_group_internal_affairs
	

	possible = {

	}

	immediate = {	
		random_country = {
			limit = {
				is_diplomatic_play_committed_participant = yes
				is_diplomatic_play_enemy_of = root
				or = {
					country_has_primary_culture = cu:yankee
					country_has_primary_culture = cu:dixie
				}
				is_diplomatic_play_target = yes
			}
		save_scope_as = civil_war_usa_free_leader
		}

	}

	complete = {
		is_at_war = no
		is_diplomatic_play_committed_participant = no
		# scope:civil_war_usa_free_leader ?= {
		# 	is_subject_of = root
		# }
		scope:civil_war_usa_free_leader ?= {
				is_subject_of = prev
		}
	}

	fail = {
		is_at_war = no
		is_diplomatic_play_committed_participant = no
		not = {
			scope:civil_war_usa_free_leader ?= {
				is_subject_of = prev
			}
		}
	}

	on_fail = {
		change_variable = {
			name = sick_us_var
			add = 1
		}
		every_country = {
			limit = {
				has_modifier = civil_war_nation
			}
			remove_modifier = civil_war_nation
		}
	}

	on_complete = {
		every_country = {
				limit = {
					has_modifier = civil_war_nation
				}
				remove_modifier = civil_war_nation
				make_independent = yes
				root = {
					create_diplomatic_pact = {
						country = prev
						type = puppet
					}
				}
				copy_laws = root
			}
	}
}
