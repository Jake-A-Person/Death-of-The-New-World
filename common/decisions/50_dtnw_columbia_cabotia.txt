clu_unite_clu = {
	is_shown = {
		country_has_primary_culture = cu:yankee
		is_subject = yes
		any_neighbouring_state = {
			owner = {
				country_has_primary_culture = cu:yankee
				is_subject = yes
				NOT = { THIS = ROOT }
				top_overlord = {
					THIS = root.top_overlord
				}
			}
		}
	}

	possible = {
		is_at_war = no
		has_technology_researched = nationalism
		top_overlord = {
			relations:root >= relations_threshold:amicable
		}
		NOT = { has_modifier = clu_test_modifier }
		custom_tooltip = {
			text = unite_clu_je_clu_tt
			any_neighbouring_state = {
				owner = {
					is_ai = yes
					relations:root >= relations_threshold:amicable
					country_has_primary_culture = cu:yankee
					is_subject = yes
					NOT = { THIS = ROOT }
					top_overlord = {
						THIS = root.top_overlord
					}
					this.gdp < root.gdp
				}
			}
		}
	}
	when_taken = {
		custom_tooltip = unite_clu_tt
		hidden_effect = {
			trigger_event = { id = can_aus.1 popup = yes }
		}
		add_modifier = {
			name = clu_test_modifier
			months = short_modifier_time
		}
	}

	ai_chance = {
		base = 0
		modifier = {
			has_technology_researched = pan-nationalism
			add = 100
		}		
	}
}

can_clu_cad_unite = {
	is_shown = {
		and = {
			exists = c:CLU
			c:CLU = {
				is_subject_of = root
			}
			exists = c:CAD
			c:CAD = {
				is_subject_of = root
			}
			exists = c:CAN
			c:CLU = {
				is_subject_of = root
			}	
		}
	}
	possible = {
		
	}
	when_taken = {
		if = {
			limit = {c:CLU = {is_player = yes}}
			c:CLU = {
				annex = c:CAD
				annex = c:CAN
				change_tag = CCC
			}
		}
		else_if = {
			limit = {c:CAN = {is_player=yes}}
			c:CAN = {
				annex = c:CLU
				annex = c:CAD
				change_tag = CCC
			}
		}
		else = {
			c:CAD = {
				annex = c:CLU
				annex = c:CAN
				change_tag = CCC
			}
		}
	}
	ai_chance = {
		base = 100
	}

}
